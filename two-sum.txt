public function twoSum($nums, $target)
    {
        $arr = [];
        for ($i = 1; $i < count($nums); $i++) {
            if (($nums[$i] + $nums[$i - 1]) == $target) {
                $arr[] = $i - 1;
                $arr[] = $i;
                break;
            }
        }
        return $arr;
    }
